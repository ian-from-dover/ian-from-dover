#!/usr/bin/env bash
source ~/.aliases

#-------------#
# bash prompt
#-------------#

export BASH_SILENCE_DEPRECATION_WARNING=1
export CLICOLOR=1
export LSCOLORS=ExFxBxDxCxegedabagacad

reset_color='\[\e[0m\]' # default colour
path_color='\[\033[32m\]' # green
git_repo_color='\[\e[91m\]' # red

parse_git_branch() {
    git branch 2> /dev/null | sed -e '/^[^*]/d' -e 's/* \(.*\)/(\1)/'
    # why is it so hard to add a single space behind
}

export PS1="${git_repo_color}\$(parse_git_branch)${path_color}\W${reset_color}> "

alias ls='ls -GFh' # -G colorises output, 
                   # -h human readable size, 
                   # -F adds / after a dir, 
                           # * after an executable, 
                           # @ after a symlink
